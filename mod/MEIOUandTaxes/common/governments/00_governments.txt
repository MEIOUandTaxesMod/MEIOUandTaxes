### TIER  1 - Emperor

despotic_monarchy_1 = {
	monarchy = yes
	
	valid_for_new_country = yes
	
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 0
			NOT = { government = administrative_monarchy_1 }
			NOT = { government = absolute_monarchy_1 }
			NOT = { government = constitutional_monarchy_1 }
			NOT = { government = enlightened_despotism_1 }
		}
		modifier = {
			factor = 0
			government = republic
		}
	}

	#bonus
	global_revolt_risk = -1
	unjustified_demands = -0.1
}
